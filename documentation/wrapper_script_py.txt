import subprocess
import sys
import os

def start_flask():
    """Start the Flask web server."""
    flask_process = subprocess.Popen([sys.executable, 'app.py'], cwd=os.path.dirname(__file__))
    return flask_process

def start_node():
    """Start the Node.js server."""
    node_process = subprocess.Popen(['node', 'server.js'], cwd=os.path.dirname(__file__))
    return node_process

if __name__ == '__main__':
    # Start both servers
    flask_process = start_flask()
    node_process = start_node()

    try:
        # Wait for both servers to complete
        flask_process.wait()
        node_process.wait()
    except KeyboardInterrupt:
        # Gracefully terminate both processes
        flask_process.terminate()
        node_process.terminate()
        print("Servers stopped.")
